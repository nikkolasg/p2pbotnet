def define name,&block
    c = Class.new(Object) do 
        class << self
            attr_accessor :fields, :dmethods
        end
        @fields = []
        @dmethods = []
        extend ClassMethod
        include InstanceMethod
        class_eval &block
    end
    AllClasses::const_set(name,c)
    c
end
