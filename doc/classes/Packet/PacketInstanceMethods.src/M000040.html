        def to_s
            str = "Packet #{self.class.name} :\n"
            str += "\t- header #{@header.to_s}\n"
            str += "\t- children size = #{children_size}\n"
            str += "\t- fields size = #{fields_size}\n"
            str += "\t- fields : \n"
            self.class.fields.each do |name,type,opts|
                val = self.send name
                str += "\t\t#{name} : #{val}\n"
            end
            children = @children.map { |c| c.to_s }.each { |str| str.split("\n").map { |line| "\t" + line }.join("\n") }.join("\n")
            str += "\t- Children :\n"
            str += children
            str
        end
